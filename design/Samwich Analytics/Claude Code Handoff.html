2. Build Order

Follow this sequence — each step unblocks the next:

  1. Tokens + globals — paste tokens.css into app/globals.css, wire Tailwind custom properties
  2. Auth — credentials provider with passphrase, middleware, private layout guard
  3. Root layout + sidebar — shell with sidebar, mobile drawer, theme toggle
  4. Portfolio page — static, no auth needed
  5. Blog — MDX files + gray-matter parsing, draft filtering by auth status
  6. Planner — time grid, quick-create dialog, daily/weekly/monthly views
  7. Notes — block editor, quick notes, notebooks (local first, then persist)

3. Auth — Passphrase Only

No Google OAuth. A server-side credentials check using NextAuth. The passphrase lives in an environment variable — never in client code.

// lib/auth.ts
import NextAuth from 'next-auth';
import Credentials from 'next-auth/providers/credentials';

export const { auth, handlers, signIn, signOut } = NextAuth({
  providers: [
    Credentials({
      credentials: { passphrase: { type: 'password' } },
      async authorize(credentials) {
        // Constant-time comparison to prevent timing attacks
        const { timingSafeEqual } = await import('node:crypto');
        const secret = Buffer.from(process.env.ADMIN_PASSPHRASE!);
        const input  = Buffer.from(credentials.passphrase as string);
        if (secret.length === input.length && timingSafeEqual(secret, input)) {
          return { id: '1', name: 'Sam Johnston', email: 'sam@samwich-analytics.com' };
        }
        return null;
      },
    }),
  ],
  pages: { signIn: '/login' },
  session: { strategy: 'jwt', maxAge: 30 * 24 * 60 * 60 }, // 30 days
});
// middleware.ts — protect all private routes
import { auth } from '@/lib/auth';
export default auth((req) => {
  const isPrivate = req.nextUrl.pathname.match(/^\/(planner|drafts|notes)/);
  if (isPrivate && !req.auth) {
    return Response.redirect(new URL('/login', req.url));
  }
});
export const config = { matcher: ['/((?!api|_next|.*\\..*).*)'] };
# .env.local
ADMIN_PASSPHRASE=your-long-random-passphrase-here
NEXTAUTH_SECRET=another-random-secret
NEXTAUTH_URL=http://localhost:3000

Security notes:

  • Use timingSafeEqual to prevent timing-based passphrase enumeration
  • Use a passphrase of 20+ characters (e.g. from openssl rand -base64 32)
  • Rate limiting: add @upstash/ratelimit on the auth endpoint — 5 attempts per 10 minutes per IP
  • HTTPS-only in production — Vercel handles this automatically
  • JWT sessions expire in 30 days — adjust maxAge as preferred

5. Blog — MDX Setup

// lib/blog.ts
import fs from 'fs';
import path from 'path';
import matter from 'gray-matter';

const BLOG_DIR = path.join(process.cwd(), 'content/blog');

export type Post = {
  slug: string; title: string; date: string;
  description: string; tags: string[]; isDraft: boolean;
};

export function getAllPosts(includesDrafts = false): Post[] {
  return fs.readdirSync(BLOG_DIR)
    .filter(f => f.endsWith('.mdx'))
    .map(f => {
      const { data } = matter(fs.readFileSync(path.join(BLOG_DIR, f), 'utf8'));
      return { slug: f.replace('.mdx',''), ...data } as Post;
    })
    .filter(p => includesDrafts || !p.isDraft)
    .sort((a,b) => new Date(b.date).getTime() - new Date(a.date).getTime());
}

export type BlockType = 'p'|'h1'|'h2'|'h3'|'bullet'|'checkbox'|'divider'|'table';

export interface Block {
  id: string; type: BlockType;
  content: string; checked?: boolean;
}

export interface QuickNote {
  id: string; title: string;
  blocks: Block[]; createdAt: string; // ISO string
}

export interface NotePage {
  id: string; title: string;
  blocks: Block[]; createdAt: string; updatedAt: string;
  parentId?: string; // notebook or section id
}

export interface Notebook {
  id: string; title: string; icon: string;
  description: string; color: string;
  children: NotePage[];
  createdAt: string;
}

const KEYS = { quickNotes: 'sw_qn', notebooks: 'sw_nb' };

export const notesStorage = {
  getQuickNotes: (): QuickNote[] =>
    JSON.parse(localStorage.getItem(KEYS.quickNotes) || '[]'),
  saveQuickNotes: (notes: QuickNote[]) =>
    localStorage.setItem(KEYS.quickNotes, JSON.stringify(notes)),
  getNotebooks: (): Notebook[] =>
    JSON.parse(localStorage.getItem(KEYS.notebooks) || '[]'),
  saveNotebooks: (nbs: Notebook[]) =>
    localStorage.setItem(KEYS.notebooks, JSON.stringify(nbs)),
};

Calendar events fetch in the planner page:

// app/(private)/planner/page.tsx
import { google } from 'googleapis';

async function getCalendarEvents(accessToken: string, date: string) {
  const calendar = google.calendar({ version: 'v3', auth: /* oauth2Client */ });
  const res = await calendar.events.list({
    calendarId: 'primary',
    timeMin: new Date(date + 'T00:00:00').toISOString(),
    timeMax: new Date(date + 'T23:59:59').toISOString(),
    singleEvents: true, orderBy: 'startTime',
  });
  return res.data.items ?? [];
}

Visual treatment: calendar events render as semi-transparent blocks behind your manual time blocks. Different stroke style (dashed border) distinguishes imported events from created blocks.

}

Better UX: on blur, switch from contenteditable to a rendered <div> with parsed HTML. On click, switch back to contenteditable with raw markdown source. This is the pattern the prototype already uses — wire it up properly with a useFocused hook per block.

Drag to reorder blocks

Use @dnd-kit/core — it handles keyboard accessibility and touch:

npm install @dnd-kit/core @dnd-kit/sortable @dnd-kit/utilities

Wrap each block in <SortableItem>. Show a 6-dot drag handle on hover (left side, same pattern as Notion). Hide handles when the block is focused for editing.

Linked pages

Add a page block type. Type [[ to trigger a search-as-you-type picker of existing notes. Store as { type: 'page', noteId: string, title: string }. Render as an olive underlined link that navigates to the referenced note.

Full-text search

For Phase 1 (localStorage): build a client-side index using Fuse.js — fuzzy search across all block content.

npm install fuse.js

For Phase 2+ (database): use Postgres full-text search (tsvector) or Typesense for instant results.
